<listing version="3.0">
import ProgressBar = laya.ui.ProgressBar;
import Handler = laya.utils.Handler;
class ProgressBar_Example {
    private progressBar: ProgressBar;
    public ProgressBar_Example() {
        Laya.init(640, 800);// Ensemble de jeu de la toile de large. 
        Laya.stage.bgColor = "#efefef";// Couleur de fond de la toile. 
        Laya.loader.load(["resource/ui/progress.png", "resource/ui/progress$bar.png"], Handler.create(this, this.onLoadComplete));// La charge de ressources. 
    }
    private onLoadComplete(): void {
        this.progressBar = new ProgressBar("resource/ui/progress.png");// Créer un  ProgressBar  Un objet d'instance de la classe  ProgressBar  . 
        this.progressBar.x = 100;// Ensemble  ProgressBar  La valeur X de propriété d'objet, des moyens de commande  ProgressBar  La position d'affichage de l'objet. 
        this.progressBar.y = 100;// Ensemble  ProgressBar  Les valeurs y des attributs d'objet, des moyens de commande  ProgressBar  La position d'affichage de l'objet. 
        this.progressBar.value = 0.3;// Ensemble  ProgressBar  Une valeur de progression. 
        this.progressBar.width = 200;// Ensemble  ProgressBar  La largeur. 
        this.progressBar.height = 50;// Ensemble  ProgressBar  En hauteur. 
        this.progressBar.sizeGrid = "5,10,5,10";// Ensemble  ProgressBar  Réseau d'informations. 
        this.progressBar.changeHandler = new Handler(this, this.onChange);// Ensemble  ProgressBar  La valeur de changement de valeur de processeur lors de l'exécution. 
        Laya.stage.addChild(this.progressBar);// Va  ProgressBar  Ajouter à la liste d'affichage. 
        Laya.timer.once(3000, this, this.changeValue);// Ensemble de 3 secondes (MS) après l'exécution de la fonction changeValue. 
    }
    private changeValue(): void {
        console.log("改变进度条的进度值。");
        this.progressBar.value = 0.6;
    }
    private onChange(value: number): void {
        console.log("进度发生改变： value=", value);
    }
}
