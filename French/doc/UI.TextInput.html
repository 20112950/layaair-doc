import Stage = laya.display.Stage;
import TextInput = laya.ui.TextInput;
import Handler = laya.utils.Handler;
class TextInput_Example {
    constructor() {
        Laya.init(640, 800);// Le mode de réglage de largeur élevé, jeu de rendre la toile. 
        Laya.stage.bgColor = "#efefef";// Couleur de fond de la toile. 
        Laya.loader.load(["resource/ui/input.png"], Handler.create(this, this.onLoadComplete));// La charge de ressources. 
    }
    private onLoadComplete(): void {
        var textInput: TextInput = new TextInput("这是一个TextInput实例。");// Créer un  TextInput  Un objet d'instance de la classe  TextInput  . 
        textInput.skin = "resource/ui/input.png";// Ensemble  TextInput  De la peau. 
        textInput.sizeGrid = "4,4,4,4";// Ensemble  TextInput  Réseau d'informations. 
        textInput.color = "#008fff";// Ensemble  TextInput  Couleur du texte. 
        textInput.font = "Arial";// Ensemble  TextInput  Police du texte. 
        textInput.bold = true;// Ensemble  TextInput  Affichage de texte en gras. 
        textInput.fontSize = 30;// Ensemble  TextInput  La taille de la police. 
        textInput.wordWrap = true;// Ensemble  TextInput  Du renvoi à la ligne automatique. 
        textInput.x = 100;// Ensemble  TextInput  La valeur X de propriété d'objet, des moyens de commande  TextInput  La position d'affichage de l'objet. 
        textInput.y = 100;// Ensemble  TextInput  Les valeurs y des attributs d'objet, des moyens de commande  TextInput  La position d'affichage de l'objet. 
        textInput.width = 300;// Ensemble  TextInput  La largeur. 
        textInput.height = 200;// Ensemble  TextInput  En hauteur. 
        Laya.stage.addChild(textInput);// Va  TextInput  Ajouter à la liste d'affichage. 
    }
}
