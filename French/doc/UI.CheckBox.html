UI. CheckBox

CheckBox Le composant présente une petite boîte à l'intérieur de la boîte peut avoir le coche. CheckBox Ensemble peut également afficher les étiquettes de texte optionnel, situé par défaut de l'étiquette CheckBox Sur la droite.

import CheckBox= laya.ui.CheckBox;
import Handler=laya.utils.Handler;
class CheckBox_Example{
    constructor()
    {
        Laya.init(640, 800);
        Laya.stage.bgColor = "#efefef";// Couleur de fond de la toile. 
        Laya.loader.load("resource/ui/check.png", Handler.create(this,this.onLoadComplete));// La charge de ressources. 
    }
    private onLoadComplete()
    {
        var checkBox:CheckBox = new CheckBox("resource/ui/check.png", "这个是一个CheckBox组件。");// Créer un  CheckBox  Un objet d'instance de la classe  CheckBox  , l'étiquette de la peau de la peau et des étiquettes entrantes. 
        checkBox.x = 100;// Ensemble  CheckBox  La valeur X de propriété d'objet, des moyens de commande  CheckBox  La position d'affichage de l'objet. 
        checkBox.y = 100;// Ensemble  CheckBox  Les valeurs y des attributs d'objet, des moyens de commande  CheckBox  La position d'affichage de l'objet. 
        checkBox.clickHandler = new Handler(this, this.onClick,[checkBox]);// Ensemble  CheckBox  Processeur d'événement de clic. 
        Laya.stage.addChild(checkBox);// Ce  CheckBox  L'objet est ajouté à la liste d'affichage. 
    }
    private onClick(checkBox:CheckBox):void
    {
        console.log("输出选中状态: checkBox.selected = " + checkBox.selected);
    }
}
