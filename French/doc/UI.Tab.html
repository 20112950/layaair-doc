<listing version="3.0">
import Tab = laya.ui.Tab;
import Handler = laya.utils.Handler;
class Tab_Example {
    constructor() {
        Laya.init(640, 800);// Ensemble de jeu de la toile de large. 
        Laya.stage.bgColor = "#efefef";// Couleur de fond de la toile. 
        Laya.loader.load(["resource/ui/tab.png"], Handler.create(this, this.onLoadComplete));// La charge de ressources. 
    }
    private onLoadComplete(): void {
        var tab: Tab = new Tab();// Créer un  Tab  Un objet d'instance de la classe  Tab  . 
        tab.skin = "resource/ui/tab.png";// Ensemble  Tab  De la peau. 
        tab.labels = "item0,item1,item2";// Ensemble  Tab  L'ensemble d'étiquettes. 
        tab.x = 100;// Ensemble  Tab  La valeur X de propriété d'objet, des moyens de commande  Tab  La position d'affichage de l'objet. 
        tab.y = 100;// Ensemble  Tab  Les valeurs y des attributs d'objet, des moyens de commande  Tab  La position d'affichage de l'objet. 
        tab.selectHandler = new Handler(this, this.onSelect);// Ensemble  Tab  Un élément de sélection de changer le processeur lors de l'exécution. 
        Laya.stage.addChild(tab);// Va  Tab  Ajouter à la liste d'affichage. 
    }
    private onSelect(index: number): void {
        console.log("当前选择的表情页索引: index= ", index);
    }
}
