<listing version="3.0">
import ComboBox = laya.ui.ComboBox;
import Handler = laya.utils.Handler;
class ComboBox_Example {
    constructor() {
        Laya.init(640, 800);// Ensemble de jeu de la toile de large. 
        Laya.stage.bgColor = "#efefef";// Couleur de fond de la toile. 
        Laya.loader.load("resource/ui/button.png", Handler.create(this, this.onLoadComplete));// La charge de ressources. 
    }
    private onLoadComplete(): void {
        console.log("资源加载完成！");
        var comboBox: ComboBox = new ComboBox("resource/ui/button.png", "item0,item1,item2,item3,item4,item5");// Créer un  ComboBox  Un objet d'instance de la classe  ComboBox  , dans la peau et l'étiquette de collecte. 
        comboBox.x = 100;// Ensemble  ComboBox  La valeur X de propriété d'objet, des moyens de commande  ComboBox  La position d'affichage de l'objet. 
        comboBox.y = 100;// Ensemble  ComboBox  La valeur X de propriété d'objet, des moyens de commande  ComboBox  La position d'affichage de l'objet. 
        comboBox.selectHandler = new Handler(this, this.onSelect);// Ensemble  ComboBox  Un processeur de sélection changement lors de l'exécution. 
        Laya.stage.addChild(comboBox);// Ce  ComboBox  L'objet est ajouté à la liste d'affichage. 
    }
    private onSelect(index: number): void {
        console.log("当前选中的项对象索引： ", index);
    }
}
