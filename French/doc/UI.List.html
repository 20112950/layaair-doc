import List = laya.ui.List;
import Handler = laya.utils.Handler;
public class List_Example {
    public List_Example() {
        Laya.init(640, 800);// Ensemble de jeu de la toile de large. 
        Laya.stage.bgColor = "#efefef";// Couleur de fond de la toile. 
        Laya.loader.load(["resource/ui/vscroll.png", "resource/ui/vscroll$bar.png", "resource/ui/vscroll$down.png", "resource/ui/vscroll$up.png"], Handler.create(this, this.onLoadComplete));
    }
    private onLoadComplete(): void {
        var arr= [];// Créer un réseau, des informations de stockage de données de la liste. 
        for (var i: number = 0; i &lt; 20; i++)
        {
            arr.push({ label: "item" + i });
        }
        var list: List = new List();// Créer un  List  Un objet d'instance de la classe  List  . 
        list.itemRender = Item;// Ensemble  List  Un dispositif de rendu de la cellule. 
        list.repeatX = 1;// Ensemble  List  Le nombre de cellules dans la direction horizontale. 
        list.repeatY = 10;// Ensemble  List  Le nombre de cellules dans la direction verticale. 
        list.vScrollBarSkin = "resource/ui/vscroll.png";// Ensemble  List  Dans la direction verticale de la barre de défilement de la peau. 
        list.array = arr;// Ensemble  List  La liste de sources de données. 
        list.pos(100, 100);// Ensemble  List  En position. 
        list.selectEnable = true;// Ensemble  List  Facultatif. 
        list.selectHandler = new Handler(this, this.onSelect);// Ensemble  List  Changer d'option d'exécution du processeur. 
        Laya.stage.addChild(list);// Va  List  Ajouter à la liste d'affichage. 
    }
    private onSelect(index: number): void {
        console.log("当前选择的项目索引： index= ", index);
    }
}
import Box = laya.ui.Box;
import Label = laya.ui.Label;
class Item extends Box {
    constructor() {
        this.graphics.drawRect(0, 0, 100, 20, null, "#ff0000");
        var label: Label = new Label();
        label.text = "100000";
        label.name = "label";// La valeur d'attribut est disposée une étiquette de nom. 
        label.size(100, 20);
        this.addChild(label);
    }
}
