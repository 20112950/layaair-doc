<listing version="3.0">
import VScrollBar = laya.ui.VScrollBar;
import Handler = laya.utils.Handler;
class VScrollBar_Example {
    private vScrollBar: VScrollBar;
    constructor() {
        Laya.init(640, 800);// Le mode de réglage de largeur élevé, jeu de rendre la toile. 
        Laya.stage.bgColor = "#efefef";// Couleur de fond de la toile. 
        Laya.loader.load(["resource/ui/vscroll.png", "resource/ui/vscroll$bar.png", "resource/ui/vscroll$down.png", "resource/ui/vscroll$up.png"], Handler.create(this, this.onLoadComplete));
    }
    private onLoadComplete(): void {
        this.vScrollBar = new VScrollBar();// Créer un  VScrollBar  HScrollBar un objet d'instance de la classe. 
        this.vScrollBar.skin = "resource/ui/vscroll.png";// Ensemble  VScrollBar  De la peau. 
        this.vScrollBar.x = 100;// Ensemble  VScrollBar  La valeur X de propriété d'objet, des moyens de commande  VScrollBar  La position d'affichage de l'objet. 
        this.vScrollBar.y = 100;// Ensemble  VScrollBar  Les valeurs y des attributs d'objet, des moyens de commande  VScrollBar  La position d'affichage de l'objet. 
        this.vScrollBar.changeHandler = new Handler(this, this.onChange);// Ensemble  VScrollBar  Le processeur de variation de roulement. 
        Laya.stage.addChild(this.vScrollBar);// Ce  VScrollBar  L'objet est ajouté à la liste d'affichage. 
    }
    private onChange(value: number): void {
        console.log("滚动条的位置： value=" + value);
    }
}
