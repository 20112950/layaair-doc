<listing version="3.0">
import ColorPicker = laya.ui.ColorPicker;
import Handler = laya.utils.Handler;
class ColorPicker_Example {
    constructor() {
        Laya.init(640, 800);// Ensemble de jeu de la toile de large. 
        Laya.stage.bgColor = "#efefef";// Couleur de fond de la toile. 
        Laya.loader.load("resource/ui/color.png", Handler.create(this, this.onLoadComplete));// La charge de ressources. 
    }
    private onLoadComplete(): void {
        console.log("资源加载完成！");
        var colorPicket: ColorPicker = new ColorPicker();// Créer un  ColorPicker  ColorPicket un objet d'instance de la classe. 
        colorPicket.skin = "resource/ui/color.png";// Disposé colorPicket de la peau. 
        colorPicket.x = 100;// La valeur de X pour définir les propriétés du colorPicket objet permettant de commander la position d'affichage de l'objet colorPicket. 
        colorPicket.y = 100;// La valeur de y régler les propriétés de l'objet colorPicket pour la commande de la position d'affichage de l'objet colorPicket. 
        colorPicket.changeHandler = new Handler(this, this.onChangeColor, [colorPicket]);// Les réglages de couleur colorPicket modifie la fonction de rappel. 
        Laya.stage.addChild(colorPicket);// Le colorPicket objet est ajouté à la liste d'affichage. 
    }
    private onChangeColor(colorPicket: ColorPicker): void {
        console.log("当前选择的颜色： " + colorPicket.selectedColor);
    }
}
