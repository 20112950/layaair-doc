import TextArea = laya.ui.TextArea;
import Handler = laya.utils.Handler;
class TextArea_Example {
    constructor() {
        Laya.init(640, 800);// Le mode de réglage de largeur élevé, jeu de rendre la toile. 
        Laya.stage.bgColor = "#efefef";// Couleur de fond de la toile. 
        Laya.loader.load(["resource/ui/input.png"], Handler.create(this, this.onLoadComplete));// La charge de ressources. 
    }
    private onLoadComplete(): void {
        var textArea: TextArea = new TextArea("这个一个TextArea实例。");// Créer un  TextArea  Un objet d'instance de la classe  TextArea  . 
        textArea.skin = "resource/ui/input.png";// Ensemble  TextArea  De la peau. 
        textArea.sizeGrid = "4,4,4,4";// Ensemble  TextArea  Réseau d'informations. 
        textArea.color = "#008fff";// Ensemble  TextArea  Couleur du texte. 
        textArea.font = "Arial";// Ensemble  TextArea  Police. 
        textArea.bold = true;// Ensemble  TextArea  Affichage de texte en gras. 
        textArea.fontSize = 20;// Ensemble  TextArea  La taille de la police du texte. 
        textArea.wordWrap = true;// Ensemble  TextArea  Du renvoi à la ligne automatique. 
        textArea.x = 100;// Ensemble  TextArea  La valeur X de propriété d'objet, des moyens de commande  TextArea  La position d'affichage de l'objet. 
        textArea.y = 100;// Ensemble  TextArea  Les valeurs y des attributs d'objet, des moyens de commande  TextArea  La position d'affichage de l'objet. 
        textArea.width = 300;// Ensemble  TextArea  La largeur. 
        textArea.height = 200;// Ensemble  TextArea  En hauteur. 
        Laya.stage.addChild(textArea);// Va  TextArea  Ajouter à la liste d'affichage. 
    }
}
